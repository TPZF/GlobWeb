 24 /*
 25 	Convert a geographic position to 3D
 26  */
 27 GlobWeb.CoordinateSystem.fromGeoTo3D = function(geo, dest)
 28 {
 29     if (!dest) { dest = new Array(3); }
 30 
 31 	var longInRad = Numeric.toRadian(geo[0]);
 32 	var latInRad = Numeric.toRadian(geo[1]);
 33 	var cosLat = Math.cos(latInRad);
 34 	
 35 	// Take height into account
 36 	var height = geo.length > 2 ? GlobWeb.CoordinateSystem.heightScale * geo[2] : 0;
 37 	var radius = GlobWeb.CoordinateSystem.radius + height;
 38 
 39     dest[0] = radius * Math.cos(longInRad) * cosLat;
 40     dest[1] = radius * Math.sin(longInRad) * cosLat;
 41     dest[2] = radius * Math.sin(latInRad);
 42 
 43     return dest;
 44 }

 48 /*
 49 	Convert a 3D position to geographic
 50     Returns 3 values [long, lat, distance from earth surface]
 51  */
 52 GlobWeb.CoordinateSystem.from3DToGeo = function(position3d, dest)
 53 {
 54     if (!dest) { dest = new Array(3); }
 55 
 56     var r = Math.sqrt(position3d[0]*position3d[0] +
 57                       position3d[1]*position3d[1] +
 58                       position3d[2]*position3d[2]);
 59     var lon = Math.atan2(position3d[1] / r, position3d[0] / r);
 60     var lat = Math.asin(position3d[2] / r);
 61 
 62     dest[0] = Numeric.toDegree(lon);
 63     dest[1] = Numeric.toDegree(lat);
 64     dest[2] = GlobWeb.CoordinateSystem.realEarthRadius * Math.abs(r - GlobWeb.CoordinateSystem.radius);
 65 
 66     return dest;
 67 }

 71 /*
 72 	Get local transformation
 73  */
 74 GlobWeb.CoordinateSystem.getLocalTransform = function(geo, dest)
 75 {
 76     if (!dest) { dest = mat4.create(); }
 77 
 78 	var longitude = geo[0] * Math.PI / 180.0;
 79 	var latitude = geo[1] * Math.PI / 180.0;
 80 	
 81 	var up = [  Math.cos(longitude)*Math.cos(latitude), Math.sin(longitude)*Math.cos(latitude), Math.sin(latitude) ];
 82 	var east = [ -Math.sin(longitude), Math.cos(longitude), 0 ];
 83 	var north = vec3.create();
 84 	vec3.cross( up, east, north );
 85 	
 86 	dest[0] = east[0];
 87 	dest[1] = east[1];
 88 	dest[2] = east[2];
 89 	dest[3] = 0.0;
 90 		
 91 	dest[4] = north[0];
 92 	dest[5] = north[1];
 93 	dest[6] = north[2];
 94 	dest[7] = 0.0;
 95 		
 96 	dest[8] = up[0];
 97 	dest[9] = up[1];
 98 	dest[10] = up[2];
 99 	dest[11] = 0.0;
100 
101 	dest[12] = 0.0;
102 	dest[13] = 0.0;
103 	dest[14] = 0.0;
104 	dest[15] = 1.0;
105 
106 	return dest;
107 }

111 /*
112 	Get local transformation
113  */
114 GlobWeb.CoordinateSystem.getLHVTransform = function(geo, dest)
115 {
116     if (!dest) { dest = mat4.create(); }
117 
118 	var longitude = geo[0] * Math.PI / 180.0;
119 	var latitude = geo[1] * Math.PI / 180.0;
120 	
121 	var up = [  Math.cos(longitude)*Math.cos(latitude), Math.sin(longitude)*Math.cos(latitude), Math.sin(latitude) ];
122 	var east = [ -Math.sin(longitude), Math.cos(longitude), 0 ];
123 	var north = vec3.create();
124 	vec3.cross( up, east, north );
125 	
126 	var pt = GlobWeb.CoordinateSystem.fromGeoTo3D(geo);
127 	
128 	dest[0] = east[0];
129 	dest[1] = east[1];
130 	dest[2] = east[2];
131 	dest[3] = 0.0;
132 		
133 	dest[4] = north[0];
134 	dest[5] = north[1];
135 	dest[6] = north[2];
136 	dest[7] = 0.0;
137 		
138 	dest[8] = up[0];
139 	dest[9] = up[1];
140 	dest[10] = up[2];
141 	dest[11] = 0.0;
142 
143 	dest[12] = pt[0];
144 	dest[13] = pt[1];
145 	dest[14] = pt[2];
146 	dest[15] = 1.0;
147 
148 	return dest;
149 }

154 	Get the side (i.e. X) vector from a local transformation
155  */
156 GlobWeb.CoordinateSystem.getSideVector = function( matrix, v )
157 {
158 	v[0] = matrix[0];
159 	v[1] = matrix[1];
160 	v[2] = matrix[2];
161 	
162     return v;
