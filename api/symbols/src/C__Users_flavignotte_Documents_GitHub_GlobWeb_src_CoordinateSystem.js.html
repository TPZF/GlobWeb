 24 /*
 25 	Convert a geographic position to 3D
 26  */
 27 GlobWeb.CoordinateSystem.fromGeoTo3D = function(geo, dest)
 28 {
 29     if (!dest) { dest = new Array(3); }
 30 
 31 	var longInRad = Numeric.toRadian(geo[0]);
 32 	var latInRad = Numeric.toRadian(geo[1]);
 33 	var cosLat = Math.cos(latInRad);
 34 	
 35 	// Take height into account
 36 	var height = geo.length > 2 ? GlobWeb.CoordinateSystem.heightScale * geo[2] : 0;
 37 	var radius = GlobWeb.CoordinateSystem.radius + height;
 38 
 39     dest[0] = radius * Math.cos(longInRad) * cosLat;
 40     dest[1] = radius * Math.sin(longInRad) * cosLat;
 41     dest[2] = radius * Math.sin(latInRad);
 42 
 43     return dest;
 44 }

 48 /*
 49 	Convert a 3D position to geographic
 50     Returns 3 values [long, lat, distance from earth surface]
 51  */
 52 GlobWeb.CoordinateSystem.from3DToGeo = function(position3d, dest)
 53 {
 54     if (!dest) { dest = new Array(3); }
 55 
 56     var r = Math.sqrt(position3d[0]*position3d[0] +
 57                       position3d[1]*position3d[1] +
 58                       position3d[2]*position3d[2]);
 59     var lon = Math.atan2(position3d[1] / r, position3d[0] / r);
 60     var lat = Math.asin(position3d[2] / r);
 61 
 62     dest[0] = Numeric.toDegree(lon);
 63     dest[1] = Numeric.toDegree(lat);
 64     dest[2] = GlobWeb.CoordinateSystem.realEarthRadius * Math.abs(r - GlobWeb.CoordinateSystem.radius);
 65 
 66     return dest;
 67 }

 71 /**
 72 *	Convert a 3D position to equatorial coordinates
 73 */
 74 GlobWeb.CoordinateSystem.from3DToEquatorial = function(position3d, dest){
 75 	
 76 	if (!dest) { dest = new Array(3); }
 77 	
 78 	var geo = [];
 79 
 80 	GlobWeb.CoordinateSystem.from3DToGeo(position3d, geo);
 81 	GlobWeb.CoordinateSystem.fromGeoToEquatorial(geo, dest);
 82 	
 83 	return dest;
 84 }

105 /**
106 *	Convert an equatorial position to geographic
107 *	@param {String[]} equatorial Array of two strings corresponding to Right Ascension and Declination
108 *					  specified by: "hours minuts seconds" and "degrees minuts seconds" respectively
109 *	@param {Float[]} dest Destination array of two floats corresponding to Longitude and Latitude
110 */
111 GlobWeb.CoordinateSystem.fromEquatorialToGeo = function(equatorial, dest){
112 	
113 	if(!dest) dest = [];
114 	
115 	// we use string because : parseFloat("-0") returns 0..
116 	function sign(stringDegree){
117 		return ((stringDegree[0] == "-") ? -1 : 1);
118 	}
119 
120 	var RA = equatorial[0].split(" ");
121 	// long
122 	var deg = parseFloat(RA[0]);
123 	var min = parseFloat(RA[1]);
124 	var sec = parseFloat(RA[2]);
125 	
126 	dest[0] = (deg + min/60 + sec/3600) * 15.;
127 	if(dest[0] > 180)
128 		dest[0] -= 360;
129 	
130 	var Decl = equatorial[1].split(" ");
131 	// lat
132 	deg = parseFloat(Decl[0]);
133 	min = parseFloat(Decl[1]);
134 	sec = parseFloat(Decl[2]);
135 	
136 	dest[1] = sign(Decl[0]) * (Math.abs(deg) + min/60 + sec/3600);
137 	
138 	return dest;
139 
140 }

144 /**
145 *	Convert a geographic position to equatorial
146 *	@param {Float[]} geo Array of two floats corresponding to Longitude and Latitude
147 *	@param {String[]} dest Destination array of two strings corresponding to Right Ascension and Declination
148 *					  specified by: 'hours+"h" minuts+"m" seconds+"s"' and 'degrees+"Â°" minuts+"\'" seconds+"""' respectively
149 * 	@see <GlobWeb.CoordinateSystem.fromDegreesToDMS>
150 * * 	@see <GlobWeb.CoordinateSystem.fromDegreesToHMS>
151 */
152 GlobWeb.CoordinateSystem.fromGeoToEquatorial = function(geo, dest){
153 	
154 	if (!dest) dest = [];
155 	
156 	var deg = geo[0];
157 	// RA
158 	if(deg < 0){
159 		deg += 360;
160 	}
161 
162 	dest[0] = GlobWeb.CoordinateSystem.fromDegreesToHMS( deg );
163 	dest[1] = GlobWeb.CoordinateSystem.fromDegreesToDMS(geo[1]);
164 	
165 	return dest;
166 }

170 /**
171  *	Function converting degrees to DMS("degrees minuts seconds")
172  * 
173  *	@param {Float} degree The degree
174  */
175 
176 GlobWeb.CoordinateSystem.fromDegreesToDMS = function(degree)
177 {
178 	function stringSign(val)
179 	{
180 		return (val>=0 ? "": "-");
181 	}
182 	
183 	var absLat = Math.abs(degree);
184 	deg = Math.floor(absLat);
185 	decimal = (absLat - deg) * 60;
186 	min = Math.floor(decimal);
187 	sec = (decimal - min) * 60;
188 	
189 	return stringSign(degree) + deg + String.fromCharCode(176) +" "+ min +"' "+ Numeric.roundNumber(sec, 2)+"\"";
190 	
191 }
192 
193 /**
194  *	Function converting degrees to HMS("hours minuts seconds")
195  *
196  *	@param {Float} degree The degree > 0
197  */
198 
199 GlobWeb.CoordinateSystem.fromDegreesToHMS = function(degree)
200 {
201 	var degree = degree/15;
202 	
203 	var absLon = Math.abs(degree);
204 	var hours = Math.floor(absLon);
205 	var decimal = (absLon - hours) * 60;
206 	var min = Math.floor(decimal);
207 	var sec = (decimal - min) * 60;
208 	
209 	return hours+"h "+min+"m "+ Numeric.roundNumber(sec, 2) +"s";
210 
211 }

215 /*
216 	Get local transformation
217  */
218 GlobWeb.CoordinateSystem.getLocalTransform = function(geo, dest)
219 {
220     if (!dest) { dest = mat4.create(); }
221 
222 	var longitude = geo[0] * Math.PI / 180.0;
223 	var latitude = geo[1] * Math.PI / 180.0;
224 	
225 	var up = [  Math.cos(longitude)*Math.cos(latitude), Math.sin(longitude)*Math.cos(latitude), Math.sin(latitude) ];
226 	var east = [ -Math.sin(longitude), Math.cos(longitude), 0 ];
227 	var north = vec3.create();
228 	vec3.cross( up, east, north );
229 	
230 	dest[0] = east[0];
231 	dest[1] = east[1];
232 	dest[2] = east[2];
233 	dest[3] = 0.0;
234 		
235 	dest[4] = north[0];
236 	dest[5] = north[1];
237 	dest[6] = north[2];
238 	dest[7] = 0.0;
239 		
240 	dest[8] = up[0];
241 	dest[9] = up[1];
242 	dest[10] = up[2];
243 	dest[11] = 0.0;
244 
245 	dest[12] = 0.0;
246 	dest[13] = 0.0;
247 	dest[14] = 0.0;
248 	dest[15] = 1.0;
249 
250 	return dest;
251 }

255 /*
256 	Get local transformation
257  */
258 GlobWeb.CoordinateSystem.getLHVTransform = function(geo, dest)
259 {
260     if (!dest) { dest = mat4.create(); }
261 
262 	var longitude = geo[0] * Math.PI / 180.0;
263 	var latitude = geo[1] * Math.PI / 180.0;
264 	
265 	var up = [  Math.cos(longitude)*Math.cos(latitude), Math.sin(longitude)*Math.cos(latitude), Math.sin(latitude) ];
266 	var east = [ -Math.sin(longitude), Math.cos(longitude), 0 ];
267 	var north = vec3.create();
268 	vec3.cross( up, east, north );
269 	
270 	var pt = GlobWeb.CoordinateSystem.fromGeoTo3D(geo);
271 	
272 	dest[0] = east[0];
273 	dest[1] = east[1];
274 	dest[2] = east[2];
275 	dest[3] = 0.0;
276 		
277 	dest[4] = north[0];
278 	dest[5] = north[1];
279 	dest[6] = north[2];
280 	dest[7] = 0.0;
281 		
282 	dest[8] = up[0];
283 	dest[9] = up[1];
284 	dest[10] = up[2];
285 	dest[11] = 0.0;
286 
287 	dest[12] = pt[0];
288 	dest[13] = pt[1];
289 	dest[14] = pt[2];
290 	dest[15] = 1.0;
291 
292 	return dest;
293 }
