 22 /** 
 23 	@constructor
 24 	Function constructor for TileWireframeLayer
 25  */
 26 GlobWeb.TileWireframeLayer = function( options )
 27 {
 28 	GlobWeb.BaseLayer.prototype.constructor.call( this, options );
 29 	this.globe = null;
 30 }
 31 
 32 /**************************************************************************************************************/
 33 
 34 GlobWeb.inherits( GlobWeb.BaseLayer,GlobWeb.TileWireframeLayer );
 35 
 36 /**************************************************************************************************************/
 37 
 38 /** 
 39   Attach the layer to the globe
 40  */
 41 GlobWeb.TileWireframeLayer.prototype._attach = function( g )
 42 {
 43 	GlobWeb.BaseLayer.prototype._attach.call( this, g );
 44 	
 45 	if ( this._visible )
 46 	{
 47 		this.globe.tileManager.addPostRenderer(this);
 48 	}
 49 	
 50 	if (!this.program)
 51 	{
 52 		var vertexShader = "\
 53 		attribute vec3 vertex;\n\
 54 		uniform mat4 modelViewMatrix;\n\
 55 		uniform mat4 projectionMatrix;\n\
 56 		void main(void) \n\
 57 		{\n\
 58 			gl_Position = projectionMatrix * modelViewMatrix * vec4(vertex, 1.0);\n\
 59 		}\n\
 60 		";
 61 
 62 		var fragmentShader = "\
 63 		precision highp float; \n\
 64 		uniform float alpha; \n\
 65 		void main(void)\n\
 66 		{\n\
 67 				gl_FragColor = vec4(1.0,1.0,1.0,alpha);\n\
 68 		}\n\
 69 		";
 70 		
 71 		this.program = new GlobWeb.Program(this.globe.renderContext);
 72 		this.program.createFromSource( vertexShader, fragmentShader );
 73 	}
 74 }

 78 /** 
 79   Detach the layer from the globe
 80  */
 81 GlobWeb.TileWireframeLayer.prototype._detach = function()
 82 {
 83 	this.globe.tileManager.removePostRenderer(this);
 84 	GlobWeb.BaseLayer.prototype._detach.call(this);
 85 }

 89 /**
 90 	Render the tiles outline
 91  */
 92 GlobWeb.TileWireframeLayer.prototype.render = function( tiles )
 93 {
 94 	var rc = this.globe.renderContext;
 95 	var gl = rc.gl;
 96 	
 97 	gl.enable(gl.BLEND);
 98 	
 99 	// Setup program
100 	this.program.apply();
101 	gl.uniformMatrix4fv(this.program.uniforms["projectionMatrix"], false, rc.projectionMatrix);
102 	
103 	var vertexAttribute = this.program.attributes['vertex'];
104 	var tileIndexBuffer = this.globe.tileManager.tileIndexBuffer;
105 	var currentIB = null;	
106 	
107 	for ( var i = 0; i < tiles.length; i++ )
108 	{
109 		var tile = tiles[i];
110 		
111 		var isLoaded = ( tile.state == GlobWeb.Tile.State.LOADED );
112 		var isLevelZero = ( tile.parentIndex == -1 );
113 	
114 		// Update uniforms for modelview matrix
115 		mat4.multiply( rc.viewMatrix, tile.matrix, rc.modelViewMatrix );
116 		gl.uniformMatrix4fv(this.program.uniforms["modelViewMatrix"], false, rc.modelViewMatrix);
117 		gl.uniform1f(this.program.uniforms["alpha"], this.opacity() );
118 			
119 		// Bind the vertex buffer
120 		gl.bindBuffer(gl.ARRAY_BUFFER, tile.vertexBuffer);
121 		gl.vertexAttribPointer(vertexAttribute, 3, gl.FLOAT, false, 4*tile.config.vertexSize, 0);
122 		
123 		var indexBuffer = ( isLoaded || isLevelZero ) ? tileIndexBuffer.getWireframe() : tileIndexBuffer.getSubWireframe(tile.parentIndex);
124 		// Bind the index buffer only if different (index buffer is shared between tiles)
125 		if ( currentIB != indexBuffer )
126 		{
127 			gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, indexBuffer);
128 			currentIB = indexBuffer;
129 		}
130 		
131 		// Draw the tiles in wireframe mode
132 		var numIndices = currentIB.numIndices;
133 		gl.drawElements(gl.LINES, currentIB.numIndices, gl.UNSIGNED_SHORT, 0);
134 	}
135 	
136 	gl.disable(gl.BLEND);
137 }

141 /**
142  * 	Set visibility of the layer
143  */
144 GlobWeb.TileWireframeLayer.prototype.visible = function( arg )
145 {
146 	if ( typeof arg == "boolean" && this._visible != arg )
147 	{
148 		this._visible = arg;
149 		
150 		if ( arg ){
151 			this.globe.tileManager.addPostRenderer(this);
152 		}
153 		else
154 		{
155 			this.globe.tileManager.removePostRenderer(this);
156 		}
157 	}
158 	
159 	return this._visible;
160 }

167 GlobWeb.TileWireframeLayer.prototype.opacity = function( arg )
168 {
