 22 /** @constructor
 23 	BoundingBox constructor
 24  */
 25 GlobWeb.BoundingBox = function(min,max)
 26 {
 27 	if (min)
 28 	{
 29 		this.min = vec3.create( min );
 30 	}
 31 	if (max)
 32 	{
 33 		this.max = vec3.create( max );
 34 	}
 35 }

 39 /**
 40 	Extent the bounding box with the given point
 41  */
 42 GlobWeb.BoundingBox.prototype.extend = function( x, y, z)
 43 {
 44 	if (!this.min)
 45 	{
 46 		this.min = vec3.create();
 47 		this.max = vec3.create();
 48 		
 49 		this.min[0] = x;
 50 		this.min[1] = y;
 51 		this.min[2] = z;
 52 		this.max[0] = x;
 53 		this.max[1] = y;
 54 		this.max[2] = z;
 55 	}
 56 	else
 57 	{
 58 		if ( x < this.min[0] )
 59 		{
 60 			this.min[0] = x;
 61 		}
 62 		if ( y < this.min[1] )
 63 		{
 64 			this.min[1] = y;
 65 		}
 66 		if ( z < this.min[2] )
 67 		{
 68 			this.min[2] = z;
 69 		}
 70 		if ( x > this.max[0] )
 71 		{
 72 			this.max[0] = x;
 73 		}
 74 		if ( y > this.max[1] )
 75 		{
 76 			this.max[1] = y;
 77 		}
 78 		if ( z > this.max[2] )
 79 		{
 80 			this.max[2] = z;
 81 		}
 82 	}
 83 }

 86 /**
 87 	Compute the bounding box from an array of vertices
 88  */
 89 GlobWeb.BoundingBox.prototype.compute = function(vertices,length,stride)
 90 {
 91 	if (!this.min)
 92 	{
 93 		this.min = vec3.create();
 94 		this.max = vec3.create();
 95 	}
 96 	
 97 	this.min[0] = vertices[0];
 98 	this.min[1] = vertices[1];
 99 	this.min[2] = vertices[2];
100 	this.max[0] = vertices[0];
101 	this.max[1] = vertices[1];
102 	this.max[2] = vertices[2];
103 	
104 	var st = stride || 3;
105 	var ll = length || vertices.length;
106 	
107 	for (var i=stride; i < ll; i += st)
108 	{
109 		for (var j=0; j < 3; j++)
110 		{
111 			if ( vertices[i+j] < this.min[j] )
112 			{
113 				this.min[j] = vertices[i+j];
114 			}
115 			if ( vertices[i+j] > this.max[j] )
116 			{
117 				this.max[j] = vertices[i+j];
118 			}
119 		}
120 	}
121 }

125 /**
126 	Get the corner of a bounding box
127  */
128 GlobWeb.BoundingBox.prototype.getCorner = function(pos)
129 {
130 	return [ pos&1 ? this.max[0] : this.min[0],
131 			pos&2 ? this.max[1] : this.min[1],
132 			pos&4 ? this.max[2] : this.min[2]	];
133 }

137 /**
138 	Get the center of a bounding box
139  */
140 GlobWeb.BoundingBox.prototype.getCenter = function()
141 {
142 	return [ (this.max[0] + this.min[0]) * 0.5,
143 			(this.max[1] + this.min[1]) * 0.5,
144 			(this.max[2] + this.min[2]) * 0.5	];
145 }

149 /**
150 	Get the radius of a bounding box
151  */
152 GlobWeb.BoundingBox.prototype.getRadius = function()
153 {
154 	var vec = vec3.create();
155 	vec3.subtract( this.max, this.min, vec)
156 	return 0.5 * vec3.length(vec);
157 }
