 22 /** 
 23 	@constructor
 24 	Function constructor for VectorLayer
 25  */
 26 GlobWeb.VectorLayer = function( options )
 27 {
 28 	GlobWeb.BaseLayer.prototype.constructor.call( this, options );
 29 	
 30 	// Set style
 31 	if ( options && options['style'] )
 32 		this.style = options['style'];
 33 	else
 34 		this.style = new GlobWeb.FeatureStyle();
 35 	
 36 	this.features = [];
 37 	this.type = "Vector";
 38 }
 39 
 40 /**************************************************************************************************************/
 41 
 42 GlobWeb.inherits( GlobWeb.BaseLayer,GlobWeb.VectorLayer );
 43 
 44 /**************************************************************************************************************/
 45 
 46 /** 
 47   Attach the vector layer to the globe
 48  */
 49 GlobWeb.VectorLayer.prototype._attach = function( g )
 50 {
 51 	GlobWeb.BaseLayer.prototype._attach.call( this, g );
 52 	
 53 	// Add the feature to renderers
 54 	for ( var i=0; i < this.features.length; i++ )
 55 	{
 56 		this._addFeatureToRenderers( this.features[i] );
 57 	}
 58 }
 59 
 60 /**************************************************************************************************************/
 61 
 62 /** 
 63   Detach the vector layer from the globe
 64  */
 65 GlobWeb.VectorLayer.prototype._detach = function()
 66 {
 67 	// Remove feature from renderers
 68 	for ( var i=0; i < this.features.length; i++ )
 69 	{
 70 		this._removeFeatureFromRenderers( this.features[i] );
 71 	}
 72 	
 73 	GlobWeb.BaseLayer.prototype._detach.call(this);
 74 }

 78 /** @export
 79   Adds a feature collection, in GeoJSON format
 80  */
 81 GlobWeb.VectorLayer.prototype.addFeatureCollection = function( featureCollection )
 82 {
 83 	// Note : use property defined as ['']  to avoid renaming when compiled in advanced mode with the closure compiler
 84 	var features = featureCollection['features'];
 85 	if ( features )
 86 	{
 87 		for ( var i = 0; i < features.length; i++)
 88 		{
 89 			this.addFeature( features[i] );
 90 		}
 91 	}
 92 }

 96 /** @export

 98 */
 99 GlobWeb.VectorLayer.prototype.removeFeatureCollection = function( featureCollection )

114 /** 
115   Add a feature to renderers
116 */
117 GlobWeb.VectorLayer.prototype._addFeatureToRenderers = function( feature )
118 {
119 	var geometry = feature['geometry']
120 	
121 	// Manage style, if undefined try with properties, otherwise use defaultStyle
122 	var style = this.style;
123 	var props = feature['properties'];
124 	if ( props && props['style'] )
125 	{
126 		style = props['style'];
127 		// Renderer hint always taken from the layer
128 		style['rendererHint'] = this.style['rendererHint'];
129 	}
130 
131 	// Manage geometry collection
132 	if ( geometry.type == "GeometryCollection" )
133 	{
134 		var geoms = geometry["geometries"];
135 		for ( var i = 0; i < geoms.length; i++ )
136 		{
137 			this.globe.vectorRendererManager.addGeometry( geoms[i], this, style );
138 		}
139 	}
140 	else
141 	{
142 		// Add geometry to renderers
143 		this.globe.vectorRendererManager.addGeometry( geometry, this, style );
144 	}
145 }

149 /** 
150   Remove a feature from renderers
151 */
152 GlobWeb.VectorLayer.prototype._removeFeatureFromRenderers = function( feature )
153 {
154 	var geometry = feature['geometry']
155 	
156 	// Manage geometry collection
157 	if ( geometry.type == "GeometryCollection" )
158 	{
159 		var geoms = geometry["geometries"];
160 		for ( var i = 0; i < geoms.length; i++ )
161 		{
162 			this.globe.vectorRendererManager.removeGeometry( geoms[i], this );
163 		}
164 	}
165 	else
166 	{
167 		this.globe.vectorRendererManager.removeGeometry( geometry, this );
168 	}
169 }

173 /** @export
174   Add a feature to the layer
175 */
176 GlobWeb.VectorLayer.prototype.addFeature = function( feature )
177 {
178 	// Check feature geometry : only add valid feature
179 	var geometry = feature['geometry'];
180 	if ( !geometry || !geometry.type )
181 		return;
182 	this.features.push( feature );
183 	
184 	// Add features to renderer if layer is attached to globe
185 	if ( this.globe )
186 	{			
187 		this._addFeatureToRenderers(feature);
188 		if (this._visible) this.globe.renderContext.requestFrame();
189 	}
190 }
191 
192 /**************************************************************************************************************/
193 
194 /** @export
195   Remove a feature from the layer
196 */
197 GlobWeb.VectorLayer.prototype.removeFeature = function( feature )
198 {
199 	var index = this.features.indexOf( feature );
200 	this.features.splice( index, 1 );
201 	if ( this.globe )
202 	{
203 		this._removeFeatureFromRenderers( feature );
204 		if (this._visible) this.globe.renderContext.requestFrame();
205 	}
206 }

210 /** @export
211   Remove all feature from the layer
212 */
213 GlobWeb.VectorLayer.prototype.removeAllFeatures = function()
214 {
215 	// Remove feature from renderers
216 	if ( this.globe )
217 	{
218 		for ( var i = 0; i < this.features.length; i++ )
219 		{
220 			this._removeFeatureFromRenderers( feature );
221 		}
222 	}
223 	this.features.length = 0;
224 	
225 	// Refresh rendering if needed
226 	if ( this.globe && this._visible )
227 	{
228 		this.globe.renderContext.requestFrame();
229 	}
230 }

234 /** @export
235   Modify feature style
236 */
237 GlobWeb.VectorLayer.prototype.modifyFeatureStyle = function( feature, style )
238 {
239 	this._removeFeatureFromRenderers( feature );
240 	feature['properties']['style'] = style;
241 	this._addFeatureToRenderers( feature );
242 }

246 /** @export
247   Modify the vector layer style
248 */
249 GlobWeb.VectorLayer.prototype.modifyStyle = function(style)
250 {
251 	for ( var i=0; i<this.features.length; i++ )
252 	{
253 		this._removeFeatureFromRenderers( this.features[i] );
254 	}
255 	
256 	this.style = style;
257 	
258 	for ( var i=0; i<this.features.length; i++ )
259 	{
260 		this._addFeatureToRenderers( this.features[i] );
261 	}
262 }
